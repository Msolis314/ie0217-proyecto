93float CDPfijaCompuesto::calcularInteres(){
94 /* Calcula el interes de un CDP con intereses fijos y calculados de forma compuesta
95 Para esto se debe dividir el plazo entre los meses del periodo de pago y multiplicar por el interes
96 Retorna el interes para el periodo de pago seleccionado */
97 int periodos = plazo / getMeses();
98 float tasaPorPeriodo;
99 switch(moneda){
100
101 case COLON:
102 /*tasaPorPeriodo es la tasa de interes por periodo de pago
103
104 La formula para calcular el interes compuesto es:
105 Interes = Capital * (1 + tasaPorPeriodo)^plazo - Capital
106 */
107
108 tasaPorPeriodo = data.Interes_Colones / (12/getMeses());
109 return capital * (pow(1 + tasaPorPeriodo, periodos) - 1);
110 case DOLAR:
111 tasaPorPeriodo = data.Interes_Dolares / (12/getMeses());
112 return capital * (pow(1 + tasaPorPeriodo, periodos) - 1);
113 break;
114 default:
115 break;
116
117 }
118 return 0;
119}

466std::string CDP::calcularFechaVencimiento(int plazo) {
467 // Calcula la fecha de vencimiento de un CDP
468 std::string fecha = (this->cliente).getFecha();
469 // fecha formato "mm/dd/yyyy"
470 std::string year = fecha.substr(6, 4);
471 std::string month = fecha.substr(0, 2);
472 std::string day = fecha.substr(3, 2);
473
474 int yearInt = std::stoi(year);
475 int monthInt = std::stoi(month);
476 int dayInt = std::stoi(day);
477
478 // a;adir el plazo al mes
479 monthInt += plazo;
480
481 // A;adir un a;o por cada 12 meses
482 while (monthInt > 12) {
483 yearInt++;
484 monthInt -= 12;
485 }
486
487 // Febrero tiene 29 dias si es a;o bisiesto
488 if (monthInt == 2) {
489 if ((yearInt % 4 == 0 && yearInt % 100 != 0) || (yearInt % 400 == 0)) {
490 if (dayInt > 29) {
491 dayInt = 29;
492 }
493 } else {
494 if (dayInt > 28) {
495 dayInt = 28;
496 }
497 }
498 }
499 // meses 30 dias
500 else if (monthInt == 4 || monthInt == 6 || monthInt == 9 || monthInt == 11) {
501 if (dayInt > 30) {
502 dayInt = 30;
503 }
504 }
505
506 // Fmm/dd/yyyy
507 std::ostringstream oss;
508 oss << std::setw(2) << std::setfill('0') << monthInt << "/"
509 << std::setw(2) << std::setfill('0') << dayInt << "/"
510 << yearInt;
511
512 return oss.str();
513}
