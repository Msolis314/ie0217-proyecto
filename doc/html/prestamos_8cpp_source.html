52// Declaración de la función calcularCuotaVariable
53float Prestamos::calcularCuotaVariable(float tasaInteres, float indiceReferencia, float capital, int plazo) {
54 float tasaMensual = (tasaInteres + indiceReferencia) / 12 / 100;
55 float factor = pow(1 + tasaMensual, plazo);
56 float cuota = capital * (tasaMensual * factor) / (factor - 1);
57 return cuota;
58}

115 } else {
116 std::cerr << "Tipo de préstamo no soportado." << std::endl;
117 sqlite3_close(db);
118 return tasaInteres;
119 }
120
121 std::string sql = "SELECT " + columna + " FROM BANKINFO ORDER BY FECHA DESC LIMIT 1";
122
123 rc = sqlite3_prepare_v2(db, sql.c_str(), -1, &stmt, NULL);
124 if (rc != SQLITE_OK) {
125 std::cerr << "Error al preparar la consulta: " << sqlite3_errmsg(db) << std::endl;
126 sqlite3_close(db);
127 return tasaInteres;
128 }
129
130 if (sqlite3_step(stmt) == SQLITE_ROW) {
131 tasaInteres = sqlite3_column_double(stmt, 0);
132 } else {
133 std::cerr << "No se encontró la tasa de interés para la combinación especificada." << std::endl;
134 }
135
136 sqlite3_finalize(stmt);
137 sqlite3_close(db);
138
139 return tasaInteres;
